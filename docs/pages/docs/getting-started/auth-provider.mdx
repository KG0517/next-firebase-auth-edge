# AuthProvider

To exchange user data between server and client, we could use custom `AuthContext` created in [previous step](/docs/getting-started/auth-context)

## Before you begin

Before copying example, make sure to setup Firebase Auth web client by following [official documentation](https://firebase.google.com/docs/web/setup)

## Example AuthProvider

The following is an example implementation of custom `AuthProvider` component that uses `AuthContext` to share user data between server and client components.

```tsx filename="AuthProvider.tsx"
"use client";

import * as React from "react";
import { getAuth, IdTokenResult, onIdTokenChanged, User as FirebaseUser } from "firebase/auth";
import { filterStandardClaims } from "next-firebase-auth-edge/lib/auth/claims";
import { AuthContext, User } from "./AuthContext";

export interface AuthProviderProps {
  serverUser: User | null;
  children: React.ReactNode;
}

function toUser(user: FirebaseUser, idTokenResult: IdTokenResult): User {
  return {
    ...user,
    customClaims: filterStandardClaims(idTokenResult.claims),
  };
}

export const AuthProvider: React.FunctionComponent<AuthProviderProps> = ({
  serverUser,
  children,
}) => {
  const [user, setUser] = React.useState(serverUser);

  const handleIdTokenChanged = async (firebaseUser: FirebaseUser | null) => {
    if (firebaseUser) {
      setUser(toUser(firebaseUser, await firebaseUser.getIdTokenResult()));
      return;
    }

    setUser(null);
  };

  React.useEffect(() => {
    return onIdTokenChanged(getAuth(), handleIdTokenChanged);
  }, []);

  return (
    <AuthContext.Provider
      value={{
        user,
      }}
    >
      {children}
    </AuthContext.Provider>
  );
};
```
